#!/bin/bash
read -p 'Enter first number:' firstNum
read -p 'Enter second number:' secondNum
addition=`expr $firstNum + $secondNum `
echo "The addition of two numbers= $addition"
subtraction=`expr $firstNum - $secondNum `
echo "The subtraction of two numbers= $subtraction"
multiplication=`expr $firstNum '*' $secondNum `
echo "The multiplication of two numbers=$multiplication"
division=`expr $firstNum / $secondNum`
echo "The division of two numbers= $division"
modulation=`expr $firstNum % $secondNum `
echo "The modulation of two numbers= $modulation"
